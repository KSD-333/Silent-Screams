Quick Reference Commands
========================

# Setup
python -m venv venv
venv\Scripts\activate          # Windows
source venv/bin/activate       # Unix
pip install -r requirements.txt

# Verify
python verify_setup.py

# Run Application
streamlit run app.py

# Test Modules
python keypoint_extractor.py
python dataset_utils.py
python models.py
python inference.py
python sound_utils.py

# Collect Data
python collect_data.py --normal_videos "./data/normal/*.mp4" --abnormal_videos "./data/abnormal/*.mp4"

# Train Model
python train_lstm.py --data_path training_data.npz --epochs 50

# Clean Up
deactivate                     # Exit virtual environment
rm -rf __pycache__            # Remove cache
rm -rf venv                   # Remove virtual environment
